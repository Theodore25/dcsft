def mean_time_to_recovery(n, N, , m, ):
    _pw = lambda x: (N - x) *  \
        if N - m <= x <= N else m *  \
        if 0 <= x < N - m else 1
    
    if n == 1:
        return 1 / _pw(0)
    
    left_term = 1 / _pw(0)
    for l in range(1, n):
        left_term *= (l * ) / _pw(l)
    
    right_term = 0
    for j in range(1, n):
        acc_prod = 1
        for l in range(j, n):
            acc_prod *= (l * ) / _pw(l)
        right_term += acc_prod / (j * )
    
    return left_term + right_term

n, N = 900, 1000
fig, ax_ = plt.subplots(figsize=(10, 6))
x = [xi for xi in range(n, N+1, 10)]

ax_.set_ylim(0, 2**6)
ax_.set_xticks(x)
ax_.set_yscale('symlog')
ax_.grid()

ax_.set(title='MEAN TIME TO RECOVERY', \
       xlabel='Number n of elementary machines in base subsystem', \
       ylabel='T(n)')

for i in (1, 2, 4, 6):
    y = [mean_time_to_recovery(ni, N, =1e-3, m=1, =i) for ni in x]
    ax_.plot(x, y, '8-', label=' = {0}'.format(i))
    ax_.legend()
plt.rcParams.update({'font.size': 14})
plt.show()
